private void button1_Click(object sender, EventArgs e)
        {
            if (string.IsNullOrWhiteSpace(textBox1.Text))
            {
                MessageBox.Show("Название темы отсутствует");
            }
            else if (themes.Items.Contains(textBox1.Text))
            {
                MessageBox.Show("Тема присутствует");
            }
            else
            {
                Word.Application app = new Word.Application();
                Word.Document doc = app.Documents.Add(Visible: true);
                doc.SaveAs($"{Environment.CurrentDirectory}\\{textBox1.Text}.docx");
                doc.Close();
                app.Quit();
                themes.Items.Add(textBox1.Text);
                tests.Add(new Test(textBox1.Text));
            }
        }
